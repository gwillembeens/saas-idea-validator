  export function generateShareUrls(title, resultUrl) {
    const shareText = `I just validated my startup idea '${title}' — scored [score]/5. Check it out:`

    const urls = {
      twitter: `https://twitter.com/intent/tweet?url=${encodeURIComponent(resultUrl)}&text=${encodeURIComponent(shareText)}`,
      linkedin: `https://www.linkedin.com/sharing/share-offsite/?url=${encodeURIComponent(resultUrl)}`,
      whatsapp: `https://wa.me/?text=${encodeURIComponent(shareText)} ${encodeURIComponent(resultUrl)}`,
    }

    return urls
  }

  Replace client/src/pages/ResultPage.jsx with full implementation:

  import { useParams, useNavigate } from 'react-router-dom'
  import { useEffect, useState } from 'react'
  import { useDispatch, useSelector } from 'react-redux'
  import { AppShell } from '../components/layout/AppShell'
  import { SignInButton } from '../components/auth/SignInButton'
  import { AuthModal } from '../components/auth/AuthModal'
  import { Button } from '../components/ui/Button'
  import { Card } from '../components/ui/Card'
  import { Scorecard } from '../components/validator/Scorecard'
  import { VerdictBadge } from '../components/validator/VerdictBadge'
  import { IdeaSummaryCard } from '../components/results/IdeaSummaryCard'
  import { CommentaryCard } from '../components/results/CommentaryCard'
  import { VerdictCard } from '../components/results/VerdictCard'
  import { parseSections } from '../utils/parseResult'
  import { parseScores } from '../utils/parseResult'
  import { generateShareUrls } from '../utils/shareUrls'
  import { setIdea } from '../store/slices/validatorSlice'
  import { Share2, Trash2, RefreshCw } from 'lucide-react'

  export function ResultPage() {
    const { id } = useParams()
    const navigate = useNavigate()
    const dispatch = useDispatch()
    const user = useSelector(s => s.auth.user)

    const [result, setResult] = useState(null)
    const [loading, setLoading] = useState(true)
    const [error, setError] = useState(null)
    const [showDeleteConfirm, setShowDeleteConfirm] = useState(false)
    const [isDeleting, setIsDeleting] = useState(false)

    useEffect(() => {
      async function fetchResult() {
        try {
          const res = await fetch(`/api/history/${id}`)
          if (!res.ok) throw new Error('Result not found')
          const data = await res.json()
          setResult(data)
        } catch (err) {
          setError(err.message)
        } finally {
          setLoading(false)
        }
      }
      fetchResult()
    }, [id])

    const handleRevalidate = () => {
      if (result) {
        dispatch(setIdea(result.idea_text))
        navigate('/')
      }
    }

    const handleDelete = async () => {
      setIsDeleting(true)
      try {
        const authHeader = user ? `Bearer ${localStorage.getItem('accessToken')}` : ''
        const res = await fetch(`/api/history/${id}`, {
          method: 'DELETE',
          headers: authHeader ? { Authorization: authHeader } : {},
        })
        if (!res.ok) throw new Error('Failed to delete')
        navigate('/history')
      } catch (err) {
        setError(err.message)
      } finally {
        setIsDeleting(false)
        setShowDeleteConfirm(false)
      }
    }

    if (loading) {
      return (
        <AppShell>
          <div className="flex items-center justify-center min-h-screen">
            <p className="font-body text-lg text-pencil">Loading...</p>
          </div>
        </AppShell>
      )
    }

    if (error) {
      return (
        <AppShell>
          <div className="flex flex-col items-center justify-center min-h-screen px-4">
            <h1 className="font-heading text-5xl text-pencil mb-4">Not Found</h1>
            <p className="font-body text-lg text-pencil">{error}</p>
            <a href="/" className="mt-6 text-blue font-body underline">← Back Home</a>
          </div>
        </AppShell>
      )
    }

    const sections = parseSections(result?.markdown_result)
    const scores = parseScores(result?.markdown_result)
    const shareUrls = generateShareUrls(result?.title, window.location.href)

    return (
      <AppShell>
        <div className="flex flex-col items-center justify-start min-h-screen px-4 py-20 md:px-8 relative">

          {/* Header with SignInButton */}
          <header className="absolute top-4 right-4 md:top-6 md:right-6">
            <SignInButton />
          </header>

          {/* Title */}
          <div className="w-full max-w-4xl text-center mb-12">
            <h1 className="font-heading text-5xl md:text-6xl text-pencil mb-2">
              {result?.title}
            </h1>
            <p className="font-body text-sm text-pencil opacity-50">
              {new Date(result?.created_at).toLocaleDateString('en-US', { month: 'long', day: 'numeric', year: 'numeric' })}
            </p>
          </div>

          {/* Split-card results */}
          <div className="w-full max-w-4xl mb-12">
            {/* Verdict Badge */}
            {scores && (
              <div className="mb-8 flex justify-center">
                <VerdictBadge />
              </div>
            )}

            {/* Idea Summary */}
            {sections?.ideaSummary && (
              <div className="mb-8">
                <IdeaSummaryCard markdown={sections.ideaSummary} />
              </div>
            )}

            {/* Scorecard */}
            {scores && (
              <div className="mb-8">
                <Card decoration="none">
                  <Scorecard />
                </Card>
              </div>
            )}

            {/* Verdict Section */}
            {sections?.verdict && (
              <div className="mb-8">
                <VerdictCard markdown={sections.verdict} />
              </div>
            )}

            {/* Commentary Section */}
            {sections?.commentary && (
              <div className="mb-8">
                <CommentaryCard markdown={sections.commentary} />
              </div>
            )}
          </div>

          {/* Action buttons */}
          <div className="w-full max-w-4xl mb-12">
            <div className="flex flex-wrap gap-4 justify-center md:justify-start">
              {/* Re-validate */}
              <Button variant="primary" onClick={handleRevalidate}>
                <RefreshCw size={18} className="mr-2 inline" />
                Re-validate This Idea
              </Button>

              {/* Share buttons */}
              <button
                onClick={() => window.open(shareUrls.twitter, '_blank')}
                className="px-4 py-3 font-body text-lg bg-blue text-white rounded hover:opacity-80 transition"
              >
                Share on X
              </button>
              <button
                onClick={() => window.open(shareUrls.linkedin, '_blank')}
                className="px-4 py-3 font-body text-lg bg-blue text-white rounded hover:opacity-80 transition"
              >
                Share on LinkedIn
              </button>
              <button
                onClick={() => window.open(shareUrls.whatsapp, '_blank')}
                className="px-4 py-3 font-body text-lg bg-blue text-white rounded hover:opacity-80 transition"
              >
                Share on WhatsApp
              </button>

              {/* Delete (owner only) */}
              {result?.isOwner && (
                <>
                  <button
                    onClick={() => setShowDeleteConfirm(true)}
                    className="px-4 py-3 font-body text-lg text-accent hover:opacity-70 transition"
                  >
                    <Trash2 size={18} className="mr-2 inline" />
                    Delete
                  </button>

                  {/* Delete Confirmation Modal */}
                  {showDeleteConfirm && (
                    <div className="fixed inset-0 flex items-center justify-center bg-black/20 z-50">
                      <Card decoration="tack" className="w-full max-w-sm p-8">
                        <h2 className="font-heading text-2xl text-pencil mb-4">Delete Result?</h2>
                        <p className="font-body text-lg text-pencil mb-6">
                          This cannot be undone.
                        </p>
                        <div className="flex gap-4">
                          <Button
                            variant="secondary"
                            onClick={() => setShowDeleteConfirm(false)}
                          >
                            Cancel
                          </Button>
                          <Button
                            variant="primary"
                            onClick={handleDelete}
                            disabled={isDeleting}
                          >
                            {isDeleting ? 'Deleting...' : 'Delete'}
                          </Button>
                        </div>
                      </Card>
                    </div>
                  )}
                </>
              )}
            </div>
          </div>

          {/* Non-owner CTA */}
          {!result?.isOwner && (
            <div className="w-full max-w-4xl text-center mb-12">
              <p className="font-body text-lg text-pencil opacity-60">
                Interested in validating your own idea?{' '}
                <a href="/" className="text-blue underline hover:text-accent">
                  Validate your startup →
                </a>
              </p>
            </div>
          )}

          {/* Footer spacing */}
          <div className="mt-20 md:mt-24" />

        </div>

        {/* Auth Modal */}
        <AuthModal />
      </AppShell>
    )
  }

  Reasoning: Public result view with split-card rendering. Re-validate resets idea in Redux and navigates home. Share buttons open intent URLs. Delete shows confirmation and requires ownership. Non-owner CTA at bottom.
  </action>
