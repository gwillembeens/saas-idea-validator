# Plan 22-02: Frontend — Analytics Components & ProfilePage Integration

## Objective

Install Recharts, create 4 analytics components (ActivityHeatmap, ScoreTrendChart, NicheBreakdown, AnalyticsSection), and render them below RevisionChains in ProfilePage. All components use the sketchbook design system (wobbly borders, hard shadows, font-heading/font-body, text-pencil). Mobile-first with horizontal scroll for heatmap.

## Must-Haves

- `recharts` installed in client/package.json
- `ActivityHeatmap` renders a 52×7 div grid using design system colors (paper → muted → pencil)
- `ScoreTrendChart` renders a Recharts LineChart — hidden if fewer than 2 data points
- `NicheBreakdown` renders CSS percentage bars with NichePill labels, sorted by count desc
- `AnalyticsSection` renders all 4 sub-sections with "Activity" heading (font-heading)
- `ProfilePage` renders `<AnalyticsSection>` below `<RevisionChains />`
- All new component tests passing
- Works at 375px mobile width

```jsx
import { useState } from 'react'

// Color scale: 0 = paper, 1-2 = muted, 3-5 = muted/pencil mix, 6+ = pencil
function getCellColor(count) {
  if (count === 0) return '#fdfbf7'       // paper
  if (count <= 2)  return '#e5e0d8'       // muted
  if (count <= 5)  return '#c8c0b4'       // muted-dark
  return '#2d2d2d'                        // pencil
}

function getCellTextColor(count) {
  return count > 5 ? '#fdfbf7' : '#2d2d2d'
}

export function ActivityHeatmap({ heatmap }) {
  const [tooltip, setTooltip] = useState(null)

  if (!heatmap || heatmap.length === 0) return null

  // Split into 52 weeks (364 days) — pad start if needed to align to Sun
  const weeks = []
  // We always have 365 days; group into columns of 7
  for (let w = 0; w < 53; w++) {
    const week = []
    for (let d = 0; d < 7; d++) {
      const idx = w * 7 + d
      if (idx < heatmap.length) {
        week.push(heatmap[idx])
      }
    }
    if (week.length > 0) weeks.push(week)
  }

  return (
    <div>
      <div
        className="overflow-x-auto pb-2"
        style={{ WebkitOverflowScrolling: 'touch' }}
      >
        <div className="flex gap-[3px]" style={{ minWidth: 'max-content' }}>
          {weeks.map((week, wi) => (
            <div key={wi} className="flex flex-col gap-[3px]">
              {week.map((day, di) => (
                <div
                  key={di}
                  className="relative"
                  style={{
                    width: 12,
                    height: 12,
                    backgroundColor: getCellColor(day.count),
                    border: '1px solid #e5e0d8',
                    borderRadius: 2,
                    cursor: day.count > 0 ? 'pointer' : 'default',
                    flexShrink: 0,
                  }}
                  onMouseEnter={() => day.count > 0 && setTooltip({ date: day.date, count: day.count })}
                  onMouseLeave={() => setTooltip(null)}
                />
              ))}
            </div>
          ))}
        </div>
      </div>

      {/* Tooltip */}
      {tooltip && (
        <p className="font-body text-xs text-pencil opacity-60 mt-1">
          {tooltip.count} validation{tooltip.count !== 1 ? 's' : ''} on {tooltip.date}
        </p>
      )}

      {/* Legend */}
      <div className="flex items-center gap-2 mt-2">
        <span className="font-body text-xs text-pencil opacity-50">Less</span>
        {[0, 2, 4, 6, 8].map(count => (
          <div
            key={count}
            style={{
              width: 10,
              height: 10,
              backgroundColor: getCellColor(count),
              border: '1px solid #e5e0d8',
              borderRadius: 2,
            }}
          />
        ))}
        <span className="font-body text-xs text-pencil opacity-50">More</span>
      </div>
    </div>
  )
}
```

    return { date: d.toISOString().slice(0, 10), count: countValue }

import {
  LineChart,
  Line,
  XAxis,
  YAxis,
  CartesianGrid,
  Tooltip,
  ResponsiveContainer,
} from 'recharts'

function CustomTooltip({ active, payload }) {
  if (!active || !payload?.length) return null
  const d = payload[0].payload
  return (
    <div
      className="bg-paper border-2 border-pencil px-3 py-2 font-body text-xs text-pencil shadow-hard"
      style={{ borderRadius: '8px 4px 8px 4px / 4px 8px 4px 8px' }}
    >
      <p className="font-heading text-sm">{d.score.toFixed(1)}/5</p>
      <p className="opacity-60 truncate max-w-[120px]">{d.title}</p>
    </div>
  )
}

export function ScoreTrendChart({ scoreTrend }) {
  // Hide if fewer than 2 data points
  if (!scoreTrend || scoreTrend.length < 2) return null

  return (
    <div style={{ width: '100%', height: 180 }}>
      <ResponsiveContainer width="100%" height="100%">
        <LineChart
          data={scoreTrend}
          margin={{ top: 8, right: 8, left: -24, bottom: 4 }}
        >
          <CartesianGrid strokeDasharray="3 3" stroke="#e5e0d8" />
          <XAxis
            dataKey="title"
            tick={false}
            axisLine={{ stroke: '#e5e0d8' }}
            tickLine={false}
          />
          <YAxis
            domain={[0, 5]}
            ticks={[0, 1, 2, 3, 4, 5]}
            tick={{ fontFamily: 'Patrick Hand, cursive', fontSize: 11, fill: '#2d2d2d' }}
            axisLine={false}
            tickLine={false}
          />
          <Tooltip content={<CustomTooltip />} />
          <Line
            type="monotone"
            dataKey="score"
            stroke="#2d5da1"
            strokeWidth={2}
            dot={{ fill: '#2d5da1', r: 3, strokeWidth: 0 }}
            activeDot={{ r: 5, fill: '#2d5da1' }}
          />
        </LineChart>
      </ResponsiveContainer>
    </div>
  )
}
```

import { NichePill } from '../ui/NichePill'

export function NicheBreakdown({ nicheBreakdown }) {
  if (!nicheBreakdown || nicheBreakdown.length === 0) return null

  const maxCount = nicheBreakdown[0].count // already sorted desc

  return (
    <div className="flex flex-col gap-3">
      {nicheBreakdown.map(({ niche, count }) => {
        const pct = maxCount > 0 ? Math.round((count / maxCount) * 100) : 0
        return (
          <div key={niche} className="flex flex-col gap-1">
            <div className="flex items-center justify-between gap-3">
              <NichePill niche={niche} size="sm" />
              <span className="font-body text-xs text-pencil opacity-60 whitespace-nowrap">
                {count} {count === 1 ? 'validation' : 'validations'}
              </span>
            </div>
            <div
              className="h-2 bg-muted"
              style={{ borderRadius: '4px 2px 4px 2px / 2px 4px 2px 4px' }}
            >
              <div
                className="h-full bg-pencil transition-all duration-300"
                style={{
                  width: `${pct}%`,
                  borderRadius: '4px 2px 4px 2px / 2px 4px 2px 4px',
                }}
              />
            </div>
          </div>
        )
      })}
    </div>
  )
}

import { ActivityHeatmap } from './ActivityHeatmap'
import { ScoreTrendChart } from './ScoreTrendChart'
import { NicheBreakdown } from './NicheBreakdown'

function StreakStats({ streaks }) {
  if (!streaks || (streaks.current === 0 && streaks.longest === 0)) return null
  return (
    <div className="flex items-center gap-6">
      <div className="flex flex-col items-center gap-0.5">
        <span className="font-heading text-2xl text-pencil leading-none">{streaks.current}</span>
        <span className="font-body text-xs text-pencil opacity-50">current streak</span>
      </div>
      <span className="text-pencil opacity-20 font-body text-lg">·</span>
      <div className="flex flex-col items-center gap-0.5">
        <span className="font-heading text-2xl text-pencil leading-none">{streaks.longest}</span>
        <span className="font-body text-xs text-pencil opacity-50">longest streak</span>
      </div>
    </div>
  )
}

function SubSection({ title, children }) {
  return (
    <div>
      <p className="font-body text-xs text-pencil opacity-50 uppercase tracking-wide mb-3">
        {title}
      </p>
      {children}
    </div>
  )
}

export function AnalyticsSection({ analytics }) {
  if (!analytics) return null

  const { heatmap, scoreTrend, nicheBreakdown, streaks } = analytics

  return (
    <div className="mt-10">
      <h2 className="font-heading text-2xl md:text-3xl text-pencil mb-6">Activity</h2>

      <div className="flex flex-col gap-8">
        {/* Heatmap */}
        {heatmap && heatmap.length > 0 && (
          <SubSection title="Validation activity">
            <ActivityHeatmap heatmap={heatmap} />
          </SubSection>
        )}

        {/* Streaks */}
        {streaks && (
          <SubSection title="Streaks">
            <StreakStats streaks={streaks} />
          </SubSection>
        )}

        {/* Score Trend */}
        {scoreTrend && scoreTrend.length >= 2 && (
          <SubSection title="Score trend">
            <ScoreTrendChart scoreTrend={scoreTrend} />
          </SubSection>
        )}

        {/* Niche Breakdown */}
        {nicheBreakdown && nicheBreakdown.length > 0 && (
          <SubSection title="Niches">
            <NicheBreakdown nicheBreakdown={nicheBreakdown} />
          </SubSection>
        )}
      </div>
    </div>
  )
}
```
