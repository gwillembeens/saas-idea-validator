```javascript
export const VERDICT_SCORES = {
  strong:     { emoji: '🟢', label: 'Strong Signal', bg: '#d1fae5', border: '#6ee7b7' },
  promising:  { emoji: '🟡', label: 'Promising',     bg: '#fef9c3', border: '#fde047' },
  needs_work: { emoji: '🟠', label: 'Needs Work',    bg: '#ffedd5', border: '#fdba74' },
  too_vague:  { emoji: '🔴', label: 'Too Vague',     bg: '#fee2e2', border: '#fca5a5' },
}

export function getVerdict(weighted) {
  if (weighted >= 4.5) return VERDICT_SCORES.strong
  if (weighted >= 3.5) return VERDICT_SCORES.promising
  if (weighted >= 2.5) return VERDICT_SCORES.needs_work
  return VERDICT_SCORES.too_vague
}
```

Create directory `client/src/constants/` if it doesn't exist.
